import { Meta, ColorPalette, ColorItem } from '@storybook/addon-docs/blocks';
import colors from "./colors.cjs"

export const colorItems = Object.keys(colors).map((color) => {
  const {hex, colorName, applications} = colors[color]
  return <div key={color}>
    <ColorItem title={color} subtitle={colorName} key={color} colors={[hex]} />
    <h3>Applications</h3>
    <ul>
      {applications.map(app => <li>{app}</li>)}
    </ul>
    <hr/>
  </div>
});

<Meta title="Colors" />


# Colors

<ColorPalette>
  { colorItems }
</ColorPalette>
